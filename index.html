Marimo notebooks are git-friendly

import numpy as np
import matplotlib.pyplot as plt

def plot(α):
    fig = plt.figure(figsize=(4, 5))
    xlim, ylim = [-np.pi, np.pi], [0, 10]
    x = np.linspace(*xlim)
    plt.plot(x, np.exp(x * α), label=r'$e^{\alpha x}$')
    plt.xlabel('$x$')
    plt.xlim(xlim)
    plt.ylim(ylim)
    plt.legend()
    plt.show()

α = 1
plot(α)
